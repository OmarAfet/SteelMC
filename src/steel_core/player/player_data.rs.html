23/// Persistent player data that can be serialized to/from NBT.
24///
25/// This structure mirrors vanilla Minecraft's player data format where possible,
26/// allowing for potential compatibility with vanilla tools.
27///
28/// # TODO: Missing vanilla fields
29/// The following fields should be added once their systems are implemented:
30/// - Food data: `foodLevel`, `foodSaturationLevel`, `foodExhaustionLevel`, `foodTickTimer`
31/// - Experience: `XpP` (progress), `XpLevel`, `XpTotal`, `XpSeed`
32/// - Active potion effects: `active_effects` (List)
33/// - Score: `Score` (Int)
34/// - Ender chest inventory: `EnderItems` (List)
35/// - Last death location: `LastDeathLocation` (`GlobalPos`)
36/// - Respawn position: `SpawnX`, `SpawnY`, `SpawnZ`, `SpawnDimension`, `SpawnForced`, `SpawnAngle`
37#[derive(Debug, Clone)]
38pub struct PersistentPlayerData {
39    /// Position (x, y, z) in absolute world coordinates.
40    /// NBT tag: `Pos` (`DoubleList`)
41    pub pos: [f64; 3],
42
43    /// Velocity (x, y, z) in blocks per tick.
44    /// NBT tag: `Motion` (`DoubleList`)
45    pub motion: [f64; 3],
46
47    /// Rotation (yaw, pitch) in degrees.
48    /// NBT tag: `Rotation` (`FloatList`)
49    pub rotation: [f32; 2],
50
51    /// Whether the player is on the ground.
52    /// NBT tag: `OnGround` (Byte)
53    pub on_ground: bool,
54
55    /// Whether the player is elytra gliding.
56    /// NBT tag: `FallFlying` (Byte)
57    pub fall_flying: bool,
58
59    /// Current health points.
60    /// NBT tag: `Health` (Float)
61    pub health: f32,
62
63    /// Current game mode (0=survival, 1=creative, 2=adventure, 3=spectator).
64    /// NBT tag: `playerGameType` (Int)
65    pub game_mode: i32,
66
67    /// Previous game mode of the player
68    /// NBT tag: `previousPlayerGameType` (Int)
69    pub prev_game_mode: i32,
70
71    /// Player abilities (flight, invulnerability, etc.).
72    /// NBT tag: `abilities` (Compound)
73    pub abilities: PersistentAbilities,
74
75    /// Inventory items with slot indices.
76    /// NBT tag: `Inventory` (List of Compounds)
77    pub inventory: Vec<PersistentSlot>,
78
79    /// Currently selected hotbar slot (0-8).
80    /// NBT tag: `SelectedItemSlot` (Int)
81    pub selected_slot: i32,
82
83    /// Dimension identifier (e.g., "minecraft:overworld").
84    /// NBT tag: `Dimension` (String)
85    pub dimension: String,
86
87    /// Data version for format migrations.
88    /// NBT tag: `DataVersion` (Int)
89    pub data_version: i32,
90}

120impl PersistentPlayerData {
121    /// Extracts persistent data from a live player.
122    #[must_use]
123    pub fn from_player(player: &Player) -> Self {
124        let pos = *player.position.lock();
125        let (yaw, pitch) = player.rotation.load();
126        let delta = *player.delta_movement.lock();
127        let abilities = player.abilities.lock();
128        let inventory = player.inventory.lock();
129        let entity_data = player.entity_data.lock();
130
131        // Collect non-empty inventory slots
132        let mut slots = Vec::new();
133        // Main inventory (0-35) and equipment (36-42)
134        for slot in 0..43 {
135            let item = inventory.get_item(slot);
136            if !item.is_empty() {
137                slots.push(PersistentSlot {
138                    slot: slot as i8,
139                    item: item.clone(),
140                });
141            }
142        }
143
144        Self {
145            pos: [pos.x, pos.y, pos.z],
146            motion: [delta.x, delta.y, delta.z],
147            rotation: [yaw, pitch],
148            on_ground: player.on_ground.load(Ordering::Relaxed),
149            fall_flying: player.fall_flying.load(Ordering::Relaxed),
150            health: *entity_data.health.get(),
151            game_mode: player.game_mode.load() as i32,
152            prev_game_mode: player.prev_game_mode.load() as i32,
153            abilities: PersistentAbilities {
154                invulnerable: abilities.invulnerable,
155                flying: abilities.flying,
156                may_fly: abilities.may_fly,
157                instabuild: abilities.instabuild,
158                may_build: abilities.may_build,
159                flying_speed: abilities.flying_speed,
160                walking_speed: abilities.walking_speed,
161            },
162            inventory: slots,
163            selected_slot: i32::from(inventory.get_selected_slot()),
164            dimension: player.world.dimension.key.to_string(),
165            data_version: PLAYER_DATA_VERSION,
166        }
167    }

169    /// Serializes the player data to an NBT compound.
170    #[must_use]
171    pub fn to_nbt(&self) -> NbtCompound {
172        let mut compound = NbtCompound::new();
173
174        // Position
175        let pos_list = NbtList::from(vec![
176            NbtTag::Double(self.pos[0]),
177            NbtTag::Double(self.pos[1]),
178            NbtTag::Double(self.pos[2]),
179        ]);
180        compound.insert("Pos", pos_list);
181
182        // Motion
183        let motion_list = NbtList::from(vec![
184            NbtTag::Double(self.motion[0]),
185            NbtTag::Double(self.motion[1]),
186            NbtTag::Double(self.motion[2]),
187        ]);
188        compound.insert("Motion", motion_list);
189
190        // Rotation
191        let rotation_list = NbtList::from(vec![
192            NbtTag::Float(self.rotation[0]),
193            NbtTag::Float(self.rotation[1]),
194        ]);
195        compound.insert("Rotation", rotation_list);
196
197        // Simple fields
198        compound.insert("OnGround", i8::from(self.on_ground));
199        compound.insert("FallFlying", i8::from(self.fall_flying));
200        compound.insert("Health", self.health);
201        compound.insert("playerGameType", self.game_mode);
202        compound.insert("SelectedItemSlot", self.selected_slot);
203        compound.insert("Dimension", self.dimension.clone());
204        compound.insert("DataVersion", self.data_version);
205
206        // Abilities compound
207        compound.insert("abilities", self.abilities.to_nbt());
208
209        // Inventory list
210        let inventory_list: Vec<NbtTag> = self
211            .inventory
212            .iter()
213            .map(|slot| {
214                let mut item_compound = match slot.item.clone().to_nbt_tag() {
215                    NbtTag::Compound(c) => c,
216                    _ => NbtCompound::new(),
217                };
218                item_compound.insert("Slot", slot.slot);
219                NbtTag::Compound(item_compound)
220            })
221            .collect();
222        compound.insert("Inventory", NbtList::from(inventory_list));
223
224        compound
225    }

227    /// Deserializes player data from an NBT compound.
228    ///
229    /// Returns `None` if required fields are missing or invalid.
230    #[must_use]
231    pub fn from_nbt(nbt: &BorrowedNbtCompound<'_>) -> Option<Self> {
232        // Convert to view type to access accessor methods
233        let nbt: NbtCompoundView<'_, '_> = nbt.into();
234        // Position
235        let pos_list = nbt.list("Pos")?;
236        let pos = [
237            pos_list.doubles()?.first().copied()?,
238            pos_list.doubles()?.get(1).copied()?,
239            pos_list.doubles()?.get(2).copied()?,
240        ];
241
242        // Motion (optional, default to zero)
243        let motion =
244            nbt.list("Motion")
245                .and_then(|l| l.doubles())
246                .map_or([0.0, 0.0, 0.0], |doubles| {
247                    [
248                        doubles.first().copied().unwrap_or(0.0),
249                        doubles.get(1).copied().unwrap_or(0.0),
250                        doubles.get(2).copied().unwrap_or(0.0),
251                    ]
252                });
253
254        // Rotation (optional, default to zero)
255        let rotation = nbt
256            .list("Rotation")
257            .and_then(|l| l.floats())
258            .map_or([0.0, 0.0], |floats| {
259                [
260                    floats.first().copied().unwrap_or(0.0),
261                    floats.get(1).copied().unwrap_or(0.0),
262                ]
263            });
264
265        // Simple fields with defaults
266        let on_ground = nbt.byte("OnGround") != Some(0);
267        let fall_flying = nbt.byte("FallFlying").is_some_and(|b| b != 0);
268        let health = nbt.float("Health").unwrap_or(20.0);
269        let game_mode = nbt.int("playerGameType").unwrap_or(0);
270        let prev_game_mode = nbt.int("previousPlayerGameType").unwrap_or(0);
271        let selected_slot = nbt.int("SelectedItemSlot").unwrap_or(0);
272        let dimension = nbt.string("Dimension").map_or_else(
273            || "minecraft:overworld".to_string(),
274            |s| s.to_str().to_string(),
275        );
276        let data_version = nbt.int("DataVersion").unwrap_or(0);
277
278        // Abilities
279        let abilities = nbt
280            .compound("abilities")
281            .map(|c| PersistentAbilities::from_nbt(&c))
282            .unwrap_or_default();
283
284        // Inventory
285        let mut inventory = Vec::new();
286        if let Some(inv_list) = nbt.list("Inventory")
287            && let Some(compounds) = inv_list.compounds()
288        {
289            for item_compound in compounds {
290                let slot = item_compound.byte("Slot").unwrap_or(0);
291                if let Some(item) = ItemStack::from_borrowed_compound(&item_compound) {
292                    inventory.push(PersistentSlot { slot, item });
293                }
294            }
295        }
296
297        Some(Self {
298            pos,
299            motion,
300            rotation,
301            on_ground,
302            fall_flying,
303            health,
304            game_mode,
305            prev_game_mode,
306            abilities,
307            inventory,
308            selected_slot,
309            dimension,
310            data_version,
311        })
312    }
313}

315impl PersistentAbilities {
316    /// Serializes abilities to an NBT compound.
317    #[must_use]
318    pub fn to_nbt(&self) -> NbtCompound {
319        let mut compound = NbtCompound::new();
320        compound.insert("invulnerable", i8::from(self.invulnerable));
321        compound.insert("flying", i8::from(self.flying));
322        compound.insert("mayfly", i8::from(self.may_fly));
323        compound.insert("instabuild", i8::from(self.instabuild));
324        compound.insert("mayBuild", i8::from(self.may_build));
325        compound.insert("flySpeed", self.flying_speed);
326        compound.insert("walkSpeed", self.walking_speed);
327        compound
328    }
329
330    /// Deserializes abilities from an NBT compound.
331    #[must_use]
332    pub fn from_nbt(nbt: &NbtCompoundView<'_, '_>) -> Self {
333        Self {
334            invulnerable: nbt.byte("invulnerable").is_some_and(|b| b != 0),
335            flying: nbt.byte("flying").is_some_and(|b| b != 0),
336            may_fly: nbt.byte("mayfly").is_some_and(|b| b != 0),
337            instabuild: nbt.byte("instabuild").is_some_and(|b| b != 0),
338            may_build: nbt.byte("mayBuild") != Some(0),
339            flying_speed: nbt.float("flySpeed").unwrap_or(0.05),
340            walking_speed: nbt.float("walkSpeed").unwrap_or(0.1),
341        }
342    }

387impl PersistentPlayerData {
388    /// Applies the saved data to a player.
389    ///
390    /// This restores position, rotation, inventory, abilities, etc.
391    pub fn apply_to_player(&self, player: &Player) {
392        use steel_utils::math::Vector3;
393
394        // Position
395        *player.position.lock() = Vector3::new(self.pos[0], self.pos[1], self.pos[2]);
396
397        // Rotation
398        player.rotation.store((self.rotation[0], self.rotation[1]));
399
400        // Motion/velocity
401        *player.delta_movement.lock() =
402            Vector3::new(self.motion[0], self.motion[1], self.motion[2]);
403
404        // Ground state
405        player.on_ground.store(self.on_ground, Ordering::Relaxed);
406        player
407            .fall_flying
408            .store(self.fall_flying, Ordering::Relaxed);
409
410        // Health
411        player.entity_data.lock().health.set(self.health);
412
413        // Game mode
414        let game_mode = self.game_mode.into();
415        player.game_mode.store(game_mode);
416
417        // Previous game mode
418        let prev_game_mode = self.prev_game_mode.into();
419        player.prev_game_mode.store(prev_game_mode);
420
421        // Abilities
422        *player.abilities.lock() = self.abilities.clone().into();
423
424        // Inventory
425        {
426            let mut inventory = player.inventory.lock();
427            // Clear existing inventory first
428            for slot in 0..43 {
429                inventory.set_item(slot, ItemStack::empty());
430            }
431            // Restore saved items
432            for slot_data in &self.inventory {
433                let slot_index = slot_data.slot as usize;
434                if slot_index < 43 {
435                    inventory.set_item(slot_index, slot_data.item.clone());
436                }
437            }
438            // Restore selected slot
439            let selected = self.selected_slot.clamp(0, 8) as u8;
440            inventory.set_selected_slot(selected);
441        }
442    }
